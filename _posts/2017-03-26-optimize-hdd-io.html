add /etc/sysctl.conf

vm.dirty_ratio = 6
vm.dirty_background_ratio = 3
vm.vfs_cache_pressure = 50
vm.swappiness = 1

load sysctl -p

add /etc/udev/rules.d/10-local.rules

ACTION=="add|change", KERNEL=="sd[a-z]", ATTR{queue/rotational}=="0", ATTR{queue/scheduler}="deadline"
